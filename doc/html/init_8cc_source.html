122 void Init::initEvalMasks() noexcept
123 {
124  using namespace EvalBB;
125  using namespace BoardUtils;
126  // initializes the file/rank masks
127  for(int32_t r = RANK_8; r >= RANK_1; --r)
128  {
129  for(int32_t f = FILE_A; f <= FILE_H; ++f)
130  {
131  int32_t sq = r * kNumFilesRanks + f;
132  FileMask[f] |= (1ULL << sq);
133  RankMask[r] |= (1ULL << sq);
134  }
135  }
136  for(uint32_t sq = 0; sq < kChessboardSize; ++sq)
137  {
138  // forward square
139  int32_t fsq = sq + kNumFilesRanks;
140  while(fsq < static_cast<signed>(kChessboardSize))
141  {
142  whitePassedMask[sq] |= (1ULL << fsq);
143  fsq += kNumFilesRanks;
144  }
145  fsq = sq - kNumFilesRanks;
146  while(fsq >= 0)
147  {
148  blackPassedMask[sq] |= (1ULL << fsq);
149  fsq -= kNumFilesRanks;
150  }
151  if(FileBrd[Sq64ToSq120[sq]] > FILE_A)
152  {
153  isolatedMask[sq] |= FileMask[FileBrd[Sq64ToSq120[sq]] - 1];
154  fsq = sq + 7;
155  while(fsq < static_cast<signed>(kChessboardSize))
156  {
157  whitePassedMask[sq] |= (1ULL << fsq);
158  fsq += kNumFilesRanks;
159  }
160  fsq = sq - 9;
161  while(fsq >= 0)
162  {
163  blackPassedMask[sq] |= (1ULL << fsq);
164  fsq -= kNumFilesRanks;
165  }
166  }
167  if(FileBrd[Sq64ToSq120[sq]] < FILE_H)
168  {
169  isolatedMask[sq] |= FileMask[FileBrd[Sq64ToSq120[sq]] + 1];
170  fsq = sq + 9;
171  while(fsq < static_cast<signed>(kChessboardSize))
172  {

174  fsq += kNumFilesRanks;
