122 Move IO::parseMove(std::string input, const Board& pos) noexcept
123 {
124  stringToLower(input);
125  if(input[1] < '1' || input[1] > '8') return NOMOVE;
126  if(input[0] < 'a' || input[0] > 'h') return NOMOVE;
127  if(input[3] < '1' || input[3] > '8') return NOMOVE;
128  if(input[2] < 'a' || input[2] > 'h') return NOMOVE;
129  uint32_t fromSq = fileRankToSq(input[0] - 'a', input[1] - '1');
130  uint32_t toSq = fileRankToSq(input[2] - 'a', input[3] - '1');
131  char promPce = input.size() > 4 ? input[4] : '*';
132  MoveList m = pos.getAllMoves();
133  for(uint32_t i = 0; i < m.size(); ++i)
134  {
135  Move cur = m[i];
136  if(cur.from() == fromSq && cur.to() == toSq)
137  {
138  if(promPce != '*')
139  {
140  uint32_t prom = cur.promoted();
141  if(PieceInfo::PieceRookQueen[prom] && !PieceInfo::PieceBishopQueen[prom] && promPce == 'r') return cur;
142  if(!PieceInfo::PieceRookQueen[prom] && PieceInfo::PieceBishopQueen[prom] && promPce == 'b') return cur;
143  if(PieceInfo::PieceRookQueen[prom] && PieceInfo::PieceBishopQueen[prom] && promPce == 'q') return cur;
144  if(PieceInfo::PieceKnight[prom] && promPce == 'n') return cur;
145  continue;
146  }
147  return cur;
148  }
149  }
150  return NOMOVE;
151 }
