91 int InputWaiting()
92 {
93 #ifndef WIN32
94  fd_set readfds;
95  struct timeval tv;
96  FD_ZERO (&readfds);
97  FD_SET (fileno(stdin), &readfds);
98  tv.tv_sec=0; tv.tv_usec=0;
99  select(16, &readfds, 0, 0, &tv);
100  return (FD_ISSET(fileno(stdin), &readfds));
101 #else
102  static int init = 0, pipe;
103  static HANDLE inh;
104  DWORD dw;
105  if (!init) {
106  init = 1;
107  inh = GetStdHandle(STD_INPUT_HANDLE);
108  pipe = !GetConsoleMode(inh, &dw);
109  if (!pipe) {
110  SetConsoleMode(inh, dw & ~(ENABLE_MOUSE_INPUT|ENABLE_WINDOW_INPUT));
111  FlushConsoleInputBuffer(inh);
112  }
113  }
114  if (pipe) {
115  if (!PeekNamedPipe(inh, NULL ,0, NULL, &dw, NULL)) return 1;
116  return dw;
117  } else {
118  GetNumberOfConsoleInputEvents(inh, &dw);
119  return dw <= 1 ? 0 : dw;
120  }

122 }
