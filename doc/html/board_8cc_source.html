263 {
264  uint32_t numAttackers = 0;
265  // Check Pawn
266  if(attacker == WHITE)
267  {
268  if(this->pieces[sq + Attack::wPCap[0]] == wP)
269  ++numAttackers;
270  if(this->pieces[sq + Attack::wPCap[1]] == wP)
271  ++numAttackers;
272  }
273  else
274  {
275  if(this->pieces[sq + Attack::bPCap[0]] == bP)
276  ++numAttackers;
277  if(this->pieces[sq + Attack::bPCap[1]] == bP)
278  ++numAttackers;
279  }
280  // Check Knight
281  uint32_t attackingKnight = attacker == WHITE ? wN : bN;
282  for(uint32_t i = 0; i < Attack::KnMoves.size(); ++i)
283  {
284  if(this->pieces[sq + Attack::KnMoves[i]] == attackingKnight)
285  ++numAttackers;
286  }
287 
288  // Check Horizontal and Vertical
289  uint32_t attackingRook = attacker == WHITE ? wR : bR;
290  uint32_t attackingQueen = attacker == WHITE ? wQ : bQ;
291  for(uint32_t i = 0 ; i < Attack::RkMoves.size(); ++i )
292  {
293  uint32_t move = Attack::RkMoves[i];
294  uint32_t t_sq = sq + move;
295  uint32_t pce = this->pieces[t_sq];
296  while(pce != OFFBOARD)
297  {
298  if(pce != EMPTY)
299  {
300  if(pce == attackingRook || pce == attackingQueen)
301  ++numAttackers;
302  break;
303  }
304  t_sq += move;
305  pce = this->pieces[t_sq];
306  }
307 
308  }
309 
310  uint32_t attackingBishop = attacker == WHITE ? wB : bB;
311  for(uint32_t i = 0 ; i < Attack::BiMoves.size(); ++i )
312  {
313  uint32_t move = Attack::BiMoves[i];
314  uint32_t t_sq = sq + move;
315  uint32_t pce = this->pieces[t_sq];
316  while(pce != OFFBOARD)
317  {
318  if(pce != EMPTY)
319  {
320  if(pce == attackingBishop || pce == attackingQueen)
321  ++numAttackers;
322  break;
323  }
324  t_sq += move;
325  pce = this->pieces[t_sq];
326  }
327  }
328 
329  uint32_t attackingKing = attacker == WHITE ? wK : bK;
330  for(uint32_t i = 0; i < Attack::KiMoves.size(); ++i)
331  {
332  if(this->pieces[sq + Attack::KiMoves[i]] == attackingKing)
333  ++numAttackers;
334  }
335  // Check Diagonals
336 
337  return numAttackers;

355 void Board::flipBoard() noexcept
356 {
357  using namespace BoardUtils;
358  std::array<int32_t, kChessboardSize> tmpPieceArray{};
359  std::array<int32_t, kNumPceTypes> inversePieceEnum {EMPTY, bP, bN, bB, bR, bQ, bK, wP, wN, wB, wR, wQ, wK};
360  int32_t tmpSide = !this->side_to_move;
361  int32_t tmpCastlePerm = 0;
362  int32_t tmpEnPas = NO_SQ;
363  if(this->castle_perm & WKCA) tmpCastlePerm |= BKCA;
364  if(this->castle_perm & WQCA) tmpCastlePerm |= BKCA;
365  if(this->castle_perm & BKCA) tmpCastlePerm |= WKCA;
366  if(this->castle_perm & BQCA) tmpCastlePerm |= WQCA;
367  if(this->en_pas != NO_SQ) tmpEnPas = Sq64ToSq120[WhiteToBlack[this->en_pas]];
368  for(uint32_t sq = 0; sq < kChessboardSize; ++sq)
369  {
370  tmpPieceArray[sq] = this->pieces[Sq64ToSq120[WhiteToBlack[sq]]];
371  }
372  this->resetBoard();
373  for(uint32_t sq = 0; sq < kChessboardSize; ++sq)
374  {
375  int32_t tmpPiece = inversePieceEnum[tmpPieceArray[sq]];
376  this->pieces[Sq64ToSq120[sq]] = tmpPiece;
377  }
378  this->side_to_move = tmpSide;
379  this->castle_perm = tmpCastlePerm;
380  this->en_pas = tmpEnPas;
381  this->pos_key = Hash::generatePosKey(*this);
382  this->updatePieceLists();
383 }
